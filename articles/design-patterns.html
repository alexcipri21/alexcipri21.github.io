1. Creational Patterns (Object Creation)

// SINGLETON
// Ensures a class has only one instance and provides a global access point
public class DatabaseConnection {
    private static DatabaseConnection instance;

    private DatabaseConnection() {} // Private constructor prevents instantiation

    public static synchronized DatabaseConnection getInstance() {
        if (instance == null) {
            instance = new DatabaseConnection();
        }
        return instance;
    }
}

// FACTORY METHOD
// Defines an interface for creating an object, letting subclasses decide the class
interface Button { void render(); }
class WindowsButton implements Button { public void render() { /* ... */ } }

abstract class Dialog {
    abstract Button createButton(); // Factory Method
    public void renderWindow() {
        Button okButton = createButton();
        okButton.render();
    }
}

2. Structural Patterns (Relationships Between Entities)

// ADAPTER
// Allows incompatible interfaces to collaborate
interface TypeC { void chargeWithTypeC(); }
interface MicroUSB { void chargeWithMicroUSB(); }

class MicroToTypeCAdapter implements TypeC {
    private MicroUSB microUsbDevice;

    public MicroToTypeCAdapter(MicroUSB device) { this.microUsbDevice = device; }
    
    @Override
    public void chargeWithTypeC() {
        microUsbDevice.chargeWithMicroUSB(); // Converts interface call
    }
}

// DECORATOR
// Dynamically attaches new responsibilities to an object without altering its structure
interface Coffee { double getCost(); }
class SimpleCoffee implements Coffee { public double getCost() { return 5.0; } }

class MilkDecorator implements Coffee {
    private Coffee coffee;
    public MilkDecorator(Coffee coffee) { this.coffee = coffee; }
    public double getCost() { return coffee.getCost() + 1.5; }
}

3. Behavioral Patterns (Communication Between Objects)

// STRATEGY
// Defines a family of algorithms, encapsulates each one, and makes them interchangeable
interface PaymentStrategy { void pay(int amount); }
class CreditCardPayment implements PaymentStrategy { public void pay(int amount) { /* ... */ } }
class PaypalPayment implements PaymentStrategy { public void pay(int amount) { /* ... */ } }

class ShoppingCart {
    private PaymentStrategy paymentMethod;
    public void setPaymentMethod(PaymentStrategy strategy) { this.paymentMethod = strategy; }
    public void checkout(int amount) { paymentMethod.pay(amount); }
}

// OBSERVER
// Subscription mechanism to notify multiple objects about any events
interface Observer { void update(String msg); }
class User implements Observer { public void update(String msg) { /* Notify user */ } }

class Newsletter {
    private List<Observer> subscribers = new ArrayList<>();
    public void subscribe(Observer o) { subscribers.add(o); }
    public void notifySubscribers() {
        for(Observer o : subscribers) o.update("New Article!");
    }
}
